//Remove duplicates optimized
class Solution {
    public int removeDuplicates(int[] nums) {
       
          int i=0;
        int j=i+1;
        while(i<nums.length && j<nums.length)
        {
            if(nums[i]==nums[j])
            {
                j++;
            }
            else
            {
                i++;
                nums[i] = nums[j];
            }
        }
        return ++i;
        
        
    }
}
